using System;
using System.Drawing;
using System.Windows.Forms;

class CanvasApp : Form
{
    public CanvasApp()
    {
        // Set the size of the form
        this.Size = new Size(400, 400);

        // Create a Paint event handler to draw on the form
        this.Paint += new PaintEventHandler(OnPaint);
    }

    private void OnPaint(object sender, PaintEventArgs e)
    {
        Graphics g = e.Graphics;

        // Draw a background rectangle
        Brush backgroundBrush = new SolidBrush(Color.FromArgb(238, 238, 238));
        g.FillRectangle(backgroundBrush, 50, 50, 300, 200);

        // Draw a line
        Pen linePen = new Pen(Color.Black);
        g.DrawLine(linePen, 100, 100, 250, 100);

        // Draw a point
        g.FillRectangle(Brushes.Black, 120, 120, 2, 2);

        // Draw a circle
        Pen circlePen = new Pen(Color.Black);
        g.DrawEllipse(circlePen, 275, 135, 50, 50);

        // Draw a rectangle
        g.FillRectangle(Brushes.Black, 150, 140, 80, 40);
    }

    public static void Main()
    {
        Application.Run(new CanvasApp());
    }
}

<script>
    const canvas = document.getElementById("myCanvas");
    const ctx = canvas.getContext("2d");

    // Draw a background rectangle
    ctx.fillStyle = "#eee";
    ctx.fillRect(50, 50, 300, 200);

    // Draw a line
    ctx.beginPath();
    ctx.moveTo(100, 100);
    ctx.lineTo(250, 100);
    ctx.stroke();

    // Draw a point
    ctx.fillStyle = "black";
    ctx.fillRect(120, 120, 2, 2);

    // Draw a circle
    ctx.beginPath();
    ctx.arc(300, 160, 25, 0, 2 * Math.PI);
    ctx.stroke();

    // Draw a rectangle
    ctx.fillStyle = "black";
    ctx.fillRect(150, 140, 80, 40);
</script>
