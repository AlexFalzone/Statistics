C# Code

          using System;

          class Program
          {
              static double[] GenerateRandomNumbers(int N)
              {
                  Random random = new Random();
                  double[] randomNumbers = new double[N];
                  
                  for (int i = 0; i < N; i++)
                  {
                      randomNumbers[i] = random.NextDouble();
                  }
          
                  return randomNumbers;
              }
          
              static int[] CalculateClassIntervals(double[] randomNumbers, int k)
              {
                  int[] intervalCounts = new int[k];
                  double intervalWidth = 1.0 / k;
          
                  for (int i = 0; i < randomNumbers.Length; i++)
                  {
                      double value = randomNumbers[i];
                      for (int j = 0; j < k; j++)
                      {
                          double lowerBound = j * intervalWidth;
                          double upperBound = (j + 1) * intervalWidth;
                          if (value >= lowerBound && value < upperBound)
                          {
                              intervalCounts[j]++;
                              break;
                          }
                      }
                  }
          
                  return intervalCounts;
              }
          
              static void Main()
              {
                  int N = 1000; // Modifica il valore di N
                  int k = 10;   // Modifica il valore di k
          
                  double[] randomNumbers = GenerateRandomNumbers(N);
                  int[] intervalCounts = CalculateClassIntervals(randomNumbers, k);
          
                  Console.WriteLine($"Random Numbers (N = {N}, k = {k}):");
                  foreach (var number in randomNumbers)
                  {
                      Console.WriteLine(number);
                  }
          
                  Console.WriteLine("Distribution into Class Intervals:");
                  for (int i = 0; i < k; i++)
                  {
                      double lowerBound = i / (double)k;
                      double upperBound = (i + 1) / (double)k;
                      Console.WriteLine($"[{lowerBound:F2}, {upperBound:F2}): {intervalCounts[i]}");
                  }
              }
          }
